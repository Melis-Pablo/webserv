## String Handling

### 1. std::string Operations
```cpp
class StringHandler {
public:
    // Finding substrings
    void string_search_examples() {
        std::string str = "Hello, World!";
        
        // find returns string::npos if not found
        size_t pos = str.find("World");
        if (pos != std::string::npos) {
            std::cout << "Found at: " << pos << std::endl;
        }
        
        // Search from end
        pos = str.rfind("o");
        
        // Find first/last of set
        pos = str.find_first_of("aeiou");
        pos = str.find_last_of("aeiou");
    }
    
    // Substring operations
    void substring_examples() {
        std::string str = "Hello, World!";
        
        // substr(pos, len)
        std::string sub = str.substr(0, 5);  // "Hello"
        
        // substring from position to end
        sub = str.substr(7);  // "World!"
    }
    
    // Modification
    void modification_examples() {
        std::string str = "Hello";
        
        // Append
        str += ", World!";
        str.append(" Welcome");
        
        // Insert
        str.insert(5, " my");
        
        // Replace
        str.replace(0, 5, "Hi");
        
        // Erase
        str.erase(0, 3);
    }
};
```

### 2. String Utilities
```cpp
class StringUtils {
public:
    // Trim whitespace
    static std::string trim(const std::string& str) {
        size_t first = str.find_first_not_of(" \t\n\r");
        size_t last = str.find_last_not_of(" \t\n\r");
        
        if (first == std::string::npos) return "";
        return str.substr(first, last - first + 1);
    }
    
    // Split string
    static std::vector<std::string> split(const std::string& str, 
                                        const std::string& delim) {
        std::vector<std::string> tokens;
        size_t prev = 0, pos = 0;
        
        while ((pos = str.find(delim, prev)) != std::string::npos) {
            tokens.push_back(str.substr(prev, pos - prev));
            prev = pos + delim.length();
        }
        
        tokens.push_back(str.substr(prev));
        return tokens;
    }
    
    // Join strings
    static std::string join(const std::vector<std::string>& vec, 
                          const std::string& delim) {
        std::string result;
        for (size_t i = 0; i < vec.size(); ++i) {
            if (i > 0) result += delim;
            result += vec[i];
        }
        return result;
    }
};
```

## File I/O

### 1. File Operations
```cpp
class FileHandler {
public:
    // Read entire file
    std::string read_file(const std::string& filename) {
        std::ifstream file(filename.c_str());
        if (!file) {
            throw std::runtime_error("Cannot open file");
        }
        
        std::stringstream buffer;
        buffer << file.rdbuf();
        return buffer.str();
    }
    
    // Write to file
    void write_file(const std::string& filename, 
                   const std::string& content) {
        std::ofstream file(filename.c_str());
        if (!file) {
            throw std::runtime_error("Cannot open file for writing");
        }
        
        file << content;
    }
    
    // Read file line by line
    std::vector<std::string> read_lines(const std::string& filename) {
        std::ifstream file(filename.c_str());
        std::vector<std::string> lines;
        std::string line;
        
        while (std::getline(file, line)) {
            lines.push_back(line);
        }
        
        return lines;
    }
};
```

### 2. Binary File Operations
```cpp
class BinaryFileHandler {
public:
    // Read binary data
    std::vector<char> read_binary(const std::string& filename) {
        std::ifstream file(filename.c_str(), 
                          std::ios::binary | std::ios::ate);
        if (!file) {
            throw std::runtime_error("Cannot open binary file");
        }
        
        std::streamsize size = file.tellg();
        file.seekg(0, std::ios::beg);
        
        std::vector<char> buffer(size);
        if (!file.read(buffer.data(), size)) {
            throw std::runtime_error("Read failed");
        }
        
        return buffer;
    }
    
    // Write binary data
    void write_binary(const std::string& filename, 
                     const std::vector<char>& data) {
        std::ofstream file(filename.c_str(), std::ios::binary);
        if (!file) {
            throw std::runtime_error("Cannot open file for writing");
        }
        
        file.write(data.data(), data.size());
    }
};
```

```cpp
class Socket {
private:
    int sockfd;

public:
    Socket(int domain, int type, int protocol) {
        sockfd = socket(domain, type, protocol);
        if (sockfd < 0) {
            throw std::runtime_error("Socket creation failed");
        }
    }
    
    ~Socket() {
        if (sockfd >= 0) {
            close(sockfd);
        }
    }
    
    // Prevent copying
    Socket(const Socket&);
    Socket& operator=(const Socket&);
    
    int get_fd() const { return sockfd; }
};
```

### 2. Smart Pointer Implementation
```cpp
template<typename T>
class AutoPtr {
private:
    T* ptr;

public:
    explicit AutoPtr(T* p = 0) : ptr(p) {}
    
    ~AutoPtr() { delete ptr; }
    
    // Transfer ownership
    AutoPtr(AutoPtr& other) : ptr(other.release()) {}
    
    AutoPtr& operator=(AutoPtr& other) {
        if (this != &other) {
            delete ptr;
            ptr = other.release();
        }
        return *this;
    }
    
    T* release() {
        T* tmp = ptr;
        ptr = 0;
        return tmp;
    }
    
    T* get() const { return ptr; }
    T& operator*() const { return *ptr; }
    T* operator->() const { return ptr; }
};
```

```cpp
class ErrorHandling {
public:
    // Resource management with RAII
    class ScopedResource {
    private:
        Resource* resource;
    
    public:
        ScopedResource() : resource(new Resource()) {}
        ~ScopedResource() { delete resource; }
    };
    
    // Error checking
    void error_checking_example() {
        // Check arguments
        if (invalid_argument) {
            throw std::invalid_argument("Invalid parameter");
        }
        
        // Check resource allocation
        Resource* res = new(std::nothrow) Resource();
        if (!res) {
            throw std::bad_alloc();
        }
        
        // Use RAII
        ScopedResource scoped_res;
        
        // Check operations
        if (operation_failed) {
            throw ServerException("Operation failed");
        }
    }
    
    // Exception safety
    class ExceptionSafe {
    private:
        Resource* resource;
        bool initialized;
    
    public:
        ExceptionSafe() : resource(0), initialized(false) {
            try {
                resource = new Resource();
                initialized = true;
            }
            catch (...) {
                cleanup();
                throw;
            }
        }
        
        ~ExceptionSafe() {
            cleanup();
        }
        
    private:
        void cleanup() {
            delete resource;
            resource = 0;
            initialized = false;
        }
    };
};
```
